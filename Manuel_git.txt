"# Projet_Academique_Gestion_Stock_medicament"
git init //Pour initier un depot git

git commit -m "first commit"
git branch -M main //pour nommer la branche
git remote add origin https://github.com/AshukaDior/Projet_Academique_Gestion_Stock_medicament.git //redirection vers le repertoire cree.
git push -u origin main

git push pour ajouter un nouveau dossier a ceux existant.
